from tkinter import *
from tkinter import ttk
window = Tk()
window.geometry("500x250+10+20")
window.title("Midterm in OOP")

label = Label(window, text = "Enter your fullname:", fg = "red")

txtfld = Entry(window, bd = 5)
txtfld.place(x=260, y=60)

button.place (x=55, y=155)

window.mainloop()
